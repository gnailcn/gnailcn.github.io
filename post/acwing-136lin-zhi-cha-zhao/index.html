AcWing 136.邻值查找

2024-02-26 链表 思维 进阶指南 AcWing 算法

原题

给定一个长度为 nnn 的序列 AAA，AAA 中的数各不相同。

对于 AAA 中的每一个数 A_iA\_iA_i，求：

min_1lej<i∣A_i−A_j∣\\min\_{1 \\le j <i}|A\_i-A\_j|min_1lej<i∣A_i−A_j∣

以及令上式取到最小值的 jjj（记为 P_iP\_iP_i）。若最小值点不唯一，则选择使 A_jA\_jA_j 较小的那个。

输入格式

第一行输入整数 nnn，代表序列长度。

第二行输入 nnn 个整数A_1…A_nA\_1…A\_nA_1…A_n,代表序列的具体数值，数值之间用空格隔开。

输出格式

输出共 n−1n-1n−1 行，每行输出两个整数，数值之间用空格隔开。

分别表示当 iii 取 2simn2 \\sim n2simn 时，对应的 min_1lej<i∣A_i−A_j∣\\min\_{1 \\le j <i}|A\_i-A\_j|min_1lej<i∣A_i−A_j∣ 和 P_iP\_iP_i 的值。

数据范围

nle105n \\le 10^5nle105,∣A_i∣le109|A\_i| \\le 10^9∣A_i∣le109

输入样例：

3
1 5 3

输出样例：

4 1
2 1

程序

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.Arrays;

public class Main
{
    static final int INF = Integer.MAX_VALUE;

    public static void main(String[] args) throws IOException
    {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        int n = Integer.parseInt(br.readLine());
        Num[] nums = new Num[n];
        Node[] idx2Node = new Node[n];
        String[] line = br.readLine().split(" ");
        for(int i = 0; i < n; i ++)
        {
            int x = Integer.parseInt(line[i]);
            nums[i] = new Num(x, i);
        }
        //排序
        Arrays.sort(nums);
        //建立双向链表(左右需要有表头节点)
        DLink dlink = new DLink();
        for(int i = 0; i < n; i ++)
        {
            //添加到链表尾部
            Num num = nums[i];
            Node newNode = dlink.addTail(num);
            idx2Node[num.idx] = newNode;
        }
        int[] ans1 = new int[n];
        int[] ans2 = new int[n];
        //在最开始的序列中（未排序）逆序计算结果
        for(int i = n-1; i >= 1; i --) //0不计算
        {
            Node node = idx2Node[i];
            int res = Integer.MAX_VALUE;
            int resIdx = -1;
            //本题题意表明一定会进去一个if语句，所以res一定会更新
            if(node.left != dlink.head)
            {
                int tmpX = Math.abs(node.value.x - node.left.value.x);
                if(tmpX < res)
                {
                    res = tmpX;
                    resIdx = node.left.value.idx;
                }
            }
            if(node.right != dlink.tail)
            {
                int tmpX = Math.abs(node.value.x - node.right.value.x);
                if(tmpX < res) //这里不能等于，因为取Aj较小的那一个(也就是前一个if语句)
                {
                    res = tmpX;
                    resIdx = node.right.value.idx;
                }
            }
            ans1[i] = res;
            ans2[i] = resIdx;
            //删除node节点
            dlink.remove(node);
        }
        for(int i = 1; i < n; i ++)
        {
            bw.write(ans1[i] + " " + (ans2[i]+1) + "\n");
        }
        bw.flush();
    }

    public static class DLink
    {
        Node head;
        Node tail;
        public DLink()
        {
            head = new Node(new Num(INF, -1));
            tail = new Node(new Num(INF, -1));
            head.right = tail;
            tail.left = head;
        }

        //尾部添加元素
        public Node addTail(Num num)
        {
            Node newNode = new Node(num, tail.left, tail);
            tail.left.right = newNode;
            tail.left = newNode;
            return newNode;
        }

        //删除节点
        public void remove(Node node)
        {
            node.left.right = node.right;
            node.right.left = node.left;
        }
    }

    public static class Node
    {
        Num value;
        Node left;
        Node right;
        public Node(Num v)
        {
            this.value = v;
            this.left = null;
            this.right = null;
        }
        public Node(Num v, Node left, Node right)
        {
            this.value = v;
            this.left = left;
            this.right = right;
        }
    }

    public static class Num implements Comparable<Num>
    {
        int x;
        int idx;
        public Num(int x, int idx)
        {
            this.x = x;
            this.idx = idx;
        }
        public int compareTo(Num b)
        {
            return this.x - b.x;
        }
    }
}
