AcWing 128. 编辑器

2024-01-23 栈 思维 进阶指南 AcWing 算法

原题

你将要实现一个功能强大的整数序列编辑器。

在开始时，序列是空的。

编辑器共有五种指令，如下：

1、I x，在光标处插入数值 xxx。
2、D，将光标前面的第一个元素删除，如果前面没有元素，则忽略此操作。
3、L，将光标向左移动，跳过一个元素，如果左边没有元素，则忽略此操作。
4、R，将光标向右移动，跳过一个元素，如果右边没有元素，则忽略此操作。
5、Q k，假设此刻光标之前的序列为 a1,a2,…,ana1,a2,…,ana_1,a_2,…,a_n，输出 max1≤i≤kSimax1≤i≤kSimax_{1 \le i \le k}S_i，其中 Si=a1+a2+…+aiSi=a1+a2+…+aiS_i=a_1+a_2+…+a_i。

输入格式

第一行包含一个整数 QQQ，表示指令的总数。

接下来 QQQ 行，每行一个指令，具体指令格式如题目描述。

输出格式

每一个 Q k 指令，输出一个整数作为结果，每个结果占一行。

数据范围

1≤Q≤1061≤Q≤1061 \le Q \le 10^6,
|x|≤103|x|≤103|x| \le 10^3,
1≤k≤n1≤k≤n1 \le k \le n

输入样例：

8
I 2
I -1
I 1
Q 3
L
D
R
Q 2

输出样例：

2
3

程序

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.Stack;
import java.util.List;
import java.util.ArrayList;

public class Main
{
    static Stack<Integer> sLeft = new Stack<Integer>();
    static Stack<Integer> sRight = new Stack<Integer>();
    static List<Integer> maxSumLeft = new ArrayList<Integer>(); //List模拟栈
    static int leftSum = 0;

    public static int top()
    {
        //获取maxSumLeft的栈顶元素
        if(!maxSumLeft.isEmpty())
        {
            return maxSumLeft.get(maxSumLeft.size() - 1);
        }
        return 0;
    }

    public static int pop()
    {
        //弹出maxSumLeft的栈顶元素
        if(!maxSumLeft.isEmpty())
        {
            return maxSumLeft.remove(maxSumLeft.size() - 1);
        }
        return 0;
    }

    public static void push(int x)
    {
        //压入maxSumLeft的栈顶元素
        maxSumLeft.add(x);
    }


    public static void main(String[] args) throws IOException
    {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        //读取第一行作为n
        int n = Integer.parseInt(br.readLine());
        for(int i = 0; i < n; i ++)
        {
            String line = br.readLine();
            char c = line.charAt(0);
            if(c == 'I')
            {
                int x = Integer.parseInt(line.substring(2));
                sLeft.push(x);
                leftSum += x;
                int maxv = leftSum;
                if(!maxSumLeft.isEmpty())
                {
                    maxv = Math.max(maxv, top());
                }
                push(maxv);
            }
            else if(c == 'D')
            {
                if(!sLeft.isEmpty())
                {
                    int x = sLeft.peek();
                    sLeft.pop();
                    leftSum -= x;
                    pop();
                }
            }
            else if(c == 'L')
            {
                if(!sLeft.isEmpty())
                {
                    int x = sLeft.pop();
                    sRight.push(x);
                    leftSum -= x;
                    pop();
                }
            }
            else if(c == 'R')
            {
                if(!sRight.isEmpty())
                {
                    int x = sRight.pop();
                    sLeft.push(x);
                    leftSum += x;
                    int maxv = leftSum;
                    if(!maxSumLeft.isEmpty())
                    {
                        maxv = Math.max(maxv, top());
                    }
                    push(maxv);
                }
            }
            else // c == 'Q'
            {
                int k = Integer.parseInt(line.substring(2));
                int ans = maxSumLeft.get(k - 1);
                bw.write(ans + "\n");
            }
            bw.flush();
        }
    }
}
