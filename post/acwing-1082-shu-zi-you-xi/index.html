AcWing 1082. 数字游戏

2024-01-20 数位DP 算法提高课 AcWing 动态规划 算法

原题

某人命名了一种不降数，这种数字必须满足从左到右各位数字呈非下降关系，如 123，446。

现在大家决定玩一个游戏，指定一个整数闭区间 [a,b]，问这个区间内有多少个不降数。

**注意：**不降数不能包含前导零。

输入格式

输入包含多组测试数据。

每组数据占一行，包含两个整数 a 和 b。

输出格式

每行给出一组测试数据的答案，即 [a,b] 之间有多少不降数。

数据范围

1≤a≤b≤231−1

输入样例：

1 9
1 19

输出样例：

9
18

程序

import java.util.Scanner;
import java.util.List;
import java.util.ArrayList;

public class Main
{
    //N:数字  M:位数
    static final int N = 12, M = 12;
    //f[i][j]表示第一位是i, 并且数的长度是j的不降数的个数
    static int[][] f = new int[N][M];
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        init(); //计算f
        while(sc.hasNext())
        {
            int a = sc.nextInt();
            int b = sc.nextInt();
            System.out.println(dp(b)-dp(a-1));
        }
    }

    //计算0~n有多少个不降数
    public static int dp(int n)
    {
        if(n == 0)
        {
            return 1;
        }
        //拆分数位, 逆序存储
        List<Integer> num = new ArrayList<Integer>();
        while(n!=0)
        {
            num.add(n%10);
            n /= 10;
        }
        int len = num.size();
        //数位dp
        int res = 0; //不降数的个数
        int last = Integer.MIN_VALUE; //上一个数位值
        for(int i = len-1; i >= 0; i --)
        {
            int x = num.get(i);
            //该位置填 0~x-1 可以得到的不降数的数量
            for(int digit = 0; digit < x; digit ++)
            {
                if(digit>=last)
                {
                    res += f[digit][i+1]; //以digit开始，总共有i+1位（后面有i位）
                }
            }
            //不能填x，结束右边的讨论分支
            if(x < last)
            {
                break;
            }
            //该位置填 x, 进入下一个位置讨论
            last = x;
            //如果已经是最后一位, 那么说明本身就是一种方案
            if(i==0)
            {
                res ++;
            }
        }
        return res;
    }

    public static void init()
    {
        //计算f
        for(int i = 0; i <= 9; i ++)
        {
            f[i][1] = 1;
        }
        for(int j = 2; j <= 10; j ++) //长度
        {
            for(int i = 0; i <= 9; i ++) //第一位数
            {
                for(int k = i; k <= 9; k ++) //下一位数
                {
                    f[i][j] += f[k][j-1];
                }
            }
        }
    }
}
