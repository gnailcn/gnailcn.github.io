leetcode 144. 二叉树的前序遍历

2023-10-22 leetcode 树 算法

给你二叉树的根节点 root ，返回它节点值的 前序 遍历。

示例 1：

输入：root = [1,null,2,3]
输出：[1,2,3]

示例 2：

输入：root = []
输出：[]

示例 3：

输入：root = [1]
输出：[1]

示例 4：

输入：root = [1,2]
输出：[1,2]

示例 5：

输入：root = [1,null,2]
输出：[1,2]

提示：

  • 树中节点数目在范围 [0, 100] 内
  • -100 <= Node.val <= 100

进阶： 递归算法很简单，你可以通过迭代算法完成吗？

程序：

  • 特殊迭代法
import java.util.Stack;
import java.util.List;
import java.util.ArrayList;

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        Stack<TreeNode> S = new Stack<TreeNode>();
        List<Integer> ans = new ArrayList<Integer>();
        if(root==null)
        {
            return ans;
        }
        S.push(root);
        while(S.size() > 0)
        {
            TreeNode node = S.pop();
            //访问根节点
            ans.add(node.val);
            //注意：先右再左
            if(node.right!=null)
            {
                S.add(node.right);
            }
            if(node.left!=null)
            {
                S.add(node.left);
            }
        }
        return ans;
    }
}
  • 一般迭代法
import java.util.Stack;
import java.util.List;
import java.util.ArrayList;

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    //万能迭代遍历法
    public List<Integer> preorderTraversal(TreeNode root) {
        Stack<TreeNode> S = new Stack<TreeNode>();
        List<Integer> ans = new ArrayList<Integer>();
        if(root==null)
        {
            return ans;
        }
        S.push(root);
        while(S.size()>0)
        {
            TreeNode top = S.pop();
            if(top==null)
            {
                //这是标记，说明现在的栈顶是访问过的元素，现在回溯回来处理结果
                TreeNode node = S.pop();
                ans.add(node.val);
            }
            else
            {
                //现在top是一个访问节点，现在需要去访问其孩子节点，并且标记top
                //右左中的顺序
                if(top.right!=null)   S.push(top.right);
                if(top.left!=null)    S.push(top.left);
                S.push(top);
                S.push(null); //标记为访问过了，等待回溯处理
            }
        }
        return ans;
    }
}
