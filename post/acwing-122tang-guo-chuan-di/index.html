AcWing 122.糖果传递

2023-12-27 思维 进阶指南 AcWing 数学 算法

原题

有 n 个小朋友坐成一圈，每人有 a[i] 个糖果。

每人只能给左右两人传递糖果。

每人每次传递一个糖果代价为 1。

求使所有人获得均等糖果的最小代价。

输入格式

第一行输入一个正整数 n，表示小朋友的个数。

接下来 n 行，每行一个整数 a[i]，表示第 i 个小朋友初始得到的糖果的颗数。

输出格式

输出一个整数，表示最小代价。

数据范围

1≤n≤1000000,
0≤a[i]≤2×109,
数据保证一定有解。

输入样例：

4
1
2
5
4

输出样例：

4

程序

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.Arrays;

public class Main
{
    public static final int N = (int)1e6+5;
    public static long[] num = new long[N];
    public static void main(String[] args) throws IOException
    {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        int n = Integer.parseInt(br.readLine());
        for(int i = 1; i <= n; i ++)
        {
            num[i] = Long.parseLong(br.readLine());
        }
        //求平均数
        long avg = 0;
        for(int i = 1; i <= n; i ++)
        {
            avg += num[i];
        }
        avg /= n;
        //最终求下面这些数的中位数
        //b-num[1], 2b-num[1]-num[2], 3b-num[1]-num[2]-num[3], ...
        //先计算前缀和
        for(int i = 1; i <= n; i ++)
        {
            num[i] += num[i-1];
        }
        //再计算上述的数
        for(int i = 1; i <= n; i ++)
        {
            num[i] = i*avg-num[i];
        }
        //求中位数, 注意本题需要证明中位数的可行性
        Arrays.sort(num, 1, n+1);
        long mid = num[(n+1)/2];
        //求最终代价
        long ans = 0;
        for(int i = 1; i <= n; i ++)
        {
            ans += Math.abs(mid-num[i]);
        }
        bw.write(ans+"\n");
        bw.flush();
    }
}
