AcWing 119. 袭击

2023-12-10 进阶指南 AcWing 待优化 递归递推 算法

原题

在与联盟的战斗中屡战屡败后，帝国撤退到了最后一个据点。

依靠其强大的防御系统，帝国击退了联盟的六波猛烈进攻。

经过几天的苦思冥想，联盟将军亚瑟终于注意到帝国防御系统唯一的弱点就是能源供应。

该系统由 NNN 个核电站供应能源，其中任何一个被摧毁都会使防御系统失效。

将军派出了 NNN 个特工进入据点之中，打算对能源站展开一次突袭。

不幸的是，由于受到了帝国空军的袭击，他们未能降落在预期位置。

作为一名经验丰富的将军，亚瑟很快意识到他需要重新安排突袭计划。

他现在最想知道的事情就是哪个特工距离其中任意一个发电站的距离最短。

你能帮他算出来这最短的距离是多少吗？

输入格式

输入中包含多组测试用例。

第一行输入整数 TTT，代表测试用例的数量。

对于每个测试用例，第一行输入整数 NNN。

接下来 NNN 行，每行输入两个整数 XXX 和 YYY，代表每个核电站的位置的 X，YX，YX，Y 坐标。

在接下来 NNN 行，每行输入两个整数 XXX 和 YYY，代表每名特工的位置的 X，YX，YX，Y 坐标。

输出格式

每个测试用例，输出一个最短距离值，结果保留三位小数。

每个输出结果占一行。

数据范围

1≤N≤1000001 \le N \le 1000001≤N≤100000,
0≤X,Y≤10000000000 \le X,Y \le 10000000000≤X,Y≤1000000000

输入样例：

2
4
0 0
0 1
1 0
1 1
2 2
2 3
3 2
3 3
4

输出样例：

1.414
0.000

程序

注意： 下面程序存在在特殊数据情况下复杂度达到O(n^2)，被hack的情况，需要优化

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.io.IOException;
import java.util.Arrays;

public class Main
{
    public static final int N = 200010;
    public static final double INF = Double.MAX_VALUE/4;
    public static Node[] nodes = new Node[N];
    public static Node[] tmp = new Node[N];
    public static void main(String[] args) throws IOException
    {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        int T = Integer.parseInt(br.readLine());
        for(int tid = 0; tid < T; tid ++)
        {
            int n = Integer.parseInt(br.readLine());
            for(int i = 0; i < n; i ++)
            {
                String[] strs = br.readLine().split(" ");
                int x = Integer.parseInt(strs[0]);
                int y = Integer.parseInt(strs[1]);
                nodes[i] = new Node(x, y, 0);
            }
            for(int i = 0; i < n; i ++)
            {
                String[] strs = br.readLine().split(" ");
                int x = Integer.parseInt(strs[0]);
                int y = Integer.parseInt(strs[1]);
                nodes[i + n] = new Node(x, y, 1);
            }
            Arrays.sort(nodes, 0, 2 * n);
            double ans = dfs(0, 2 * n - 1);
            bw.write(String.format("%.3f\n", ans));
            bw.flush();
        }
    }

    public static class Node implements Comparable<Node>
    {
        int x, y;
        int type;
        Node(int x, int y, int type)
        {
            this.x = x;
            this.y = y;
            this.type = type;
        }

        //按照x排序
        @Override
        public int compareTo(Node o)
        {
            return this.x - o.x;
        }
    }

    //返回x在[l, r]区间上的点集中的最短点对距离
    //对y进行归并排序
    public static double dfs(int l, int r)
    {
        if(l==r)
        {
            return INF;
        }
        int mid = (l + r) >> 1;
        double midX = nodes[mid].x;
        double d = Math.min(dfs(l, mid), dfs(mid + 1, r));
        int tpi = l, tpj = mid+1, tpk = 0;
        while(tpi <= mid && tpj <= r)
        {
            if(nodes[tpi].y < nodes[tpj].y)
            {
                tmp[tpk++] = nodes[tpi++];
            }
            else
            {
                tmp[tpk++] = nodes[tpj++];
            }
        }
        while(tpi <= mid)
        {
            tmp[tpk++] = nodes[tpi++];
        }
        while(tpj <= r)
        {
            tmp[tpk++] = nodes[tpj++];
        }
        for(int i = 0; i < tpk; i ++)
        {
            nodes[l + i] = tmp[i];
        }
        //求在[midX- d, midX + d]区间上的最短点集合，放入tmp中
        tpk = 0;
        double ans = d;
        //求tmp中的最短点对距离
        for(int i = 0; i < tpk; i ++)
        {
            for(int j = i+1; j < tpk; j ++)
            {
                if(Math.abs(tmp[i].y - tmp[j].y) >= d)
                {
                    break; //直接跳出循环，因为y排好序了，后面的点对距离肯定大于d
                }
                double dis = getDistance(tmp[i], tmp[j]);
                ans = Math.min(ans, dis);
            }
        }
        return ans;
    }

    public static double getDistance(Node a, Node b)
    {
        if(a.type == b.type)
        {
            return INF;
        }
        return Math.sqrt((a.x - b.x)*(a.x - b.x) + (a.y - b.y)*(a.y - b.y));
    }
}
