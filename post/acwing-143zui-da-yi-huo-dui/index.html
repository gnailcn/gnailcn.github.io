AcWing 143.最大异或对

2024-03-08 Trie 进阶指南 AcWing 算法

原题

在给定的 N 个整数 A_1，A_2……A_N 中选出两个进行 xor（异或）运算，得到的结果最大是多少？

输入格式

第一行输入一个整数 N。

第二行输入 N 个整数 A_1～A_N。

输出格式

输出一个整数表示答案。

数据范围

1 \le N \le 10^5,
0 \le A_i < 2^{31}

输入样例：

3
1 2 3

输出样例：

3

程序

import java.util.Scanner;

public class Main
{
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        Trie trie = new Trie();
        int n = sc.nextInt();
        int[] nums = new int[n];
        for(int i = 0; i < n; i++)
        {
            nums[i] = sc.nextInt();
            trie.insert(nums[i]);
        }
        int ans = 0;
        for(int i = 0; i < n; i++)
        {
            ans = Math.max(ans, trie.findMaxXOR(nums[i]));
        }
        System.out.println(ans);
    }

    //Trie Tree
    public static class Trie
    {
        Node root;
        public Trie()
        {
            root = new Node(-1);
        }
        public void insert(int num)
        {
            Node cur = root;
            for(int i = 31; i >= 0; i--)
            {
                int bit = (num >> i) & 1;
                if(cur.next[bit] == null)
                {
                    cur.next[bit] = new Node(bit);
                }
                cur = cur.next[bit];
            }
        }
        public int findMaxXOR(int num)
        {
            Node cur = root;
            int res = 0;
            for(int i = 31; i >= 0; i--)
            {
                int bit = (num >> i) & 1;
                //1找0，0找1
                if(cur.next[bit ^ 1] != null) //找到了
                {
                    res += (1 << i); //加上这一位的异或结果
                    cur = cur.next[bit ^ 1];
                }
                else //没找到
                {
                    //异或结果为0，所以不用加
                    cur = cur.next[bit];
                }
            }
            return res;
        }
    }

    //trie树的节点
    public static class Node
    {
        int value; // 0 or 1
        boolean isEnd; // 是否是一个数的结尾， 即前面的01可以组成一个数（二进制）
        Node[] next = new Node[2]; // 0 or 1
        public Node(int v)
        {
            value = v;
            isEnd = false;
            for(int i = 0; i < 2; i++)
            {
                next[i] = null;
            }
        }
    }
}
