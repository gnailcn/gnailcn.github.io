AcWing 785.快速排序

2024-03-15 算法基础课 待完善 AcWing 排序 递归递推 算法

原题

问题描述

给定一个长度为 n 的整数数列，请你使用快速排序对这个数列按照从小到大进行排序，并将排好序的数列按顺序输出。

输入格式

输入共两行，第一行包含整数 n。

第二行包含 n 个整数（所有整数均在 1 到 10^9 范围内），表示整个数列。

输出格式

输出共一行，包含 n 个整数，表示排好序的数列。

数据范围

1 ≤ n ≤ 100000

输入样例：

5
3 1 2 4 5

输出样例：

1 2 3 4 5

程序

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.io.IOException;

public class Main
{
    static int n;
    static int[] nums;
    public static void main(String[] args) throws IOException
    {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        n = Integer.parseInt(br.readLine());
        nums = new int[n];
        String[] line2 = br.readLine().split(" ");
        for (int i = 0; i < n; i++)
        {
            nums[i] = Integer.parseInt(line2[i]);
        }
        quickSort(0, n-1);
        for (int i = 0; i < n; i++)
        {
            bw.write(nums[i]+" ");
        }
        bw.write("\n");
        bw.flush();
    }
    static void quickSort(int l, int r)
    {
        if(l>=r)
        {
            return;
        }
        int x = nums[(l+r)/2]; //可以随机取值
        int i = l-1, j = r+1;
        while(i<j)
        {
            //下面i和j的移动不加等号判断是为了防止越过数组下标
            do i++; while(nums[i]<x);
            do j--; while(nums[j]>x);
            //额外添加判断是为了防止i和j刚好交叉的情况
            if(i<j)
            {
                int t = nums[i];
                nums[i] = nums[j];
                nums[j] = t;
            }
        }
        //最终i==j或者i==j+1
        int bound = j;
        quickSort(l, bound);
        quickSort(bound+1, r);
    }
}
