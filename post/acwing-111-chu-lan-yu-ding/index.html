AcWing 111. 畜栏预定

2023-12-10 贪心 进阶指南 AcWing 堆 算法

原题

有 NNN 头牛在畜栏中吃草。

每个畜栏在同一时间段只能提供给一头牛吃草，所以可能会需要多个畜栏。

给定 NNN 头牛和每头牛开始吃草的时间 AAA 以及结束吃草的时间 BBB，每头牛在 [A,B][A,B][A,B] 这一时间段内都会一直吃草。

当两头牛的吃草区间存在交集时（包括端点），这两头牛不能被安排在同一个畜栏吃草。

求需要的最小畜栏数目和每头牛对应的畜栏方案。

输入格式

第 111 行：输入一个整数 NNN。

第 2..N+12..N+12..N+1 行：第 i+1i+1i+1 行输入第 iii 头牛的开始吃草时间 AAA 以及结束吃草时间 BBB，数之间用空格隔开。

输出格式

第 111 行：输出一个整数，代表所需最小畜栏数。

第 2..N+12..N+12..N+1 行：第 i+1i+1i+1 行输出第 iii 头牛被安排到的畜栏编号，编号是从 111 开始的 连续 整数，只要方案合法即可。

数据范围

1≤N≤500001 \le N \le 500001≤N≤50000,
1≤A,B≤10000001 \le A,B \le 10000001≤A,B≤1000000

输入样例：

5
1 10
2 4
3 6
5 8
4 7

输出样例：

4
1
2
3
2
4

程序

//导入快读和快写的包
import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.Arrays;
import java.util.PriorityQueue;

public class Main
{
    public static final int N = 50005;
    public static Interval[] interval = new Interval[N];

    public static int[] ans = new int[N];
    public static void main(String[] args) throws IOException
    {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        //第一行输入一个整数n
        int n = Integer.parseInt(br.readLine());
        //接下来n行，每行两个整数，分别是a和b
        for(int i = 0; i < n; i ++)
        {
            String[] line = br.readLine().split(" ");
            interval[i] = new Interval(i, Integer.parseInt(line[0]), Integer.parseInt(line[1]));
        }
        Arrays.sort(interval, 0, n);
        //将所有牛栏的右端点存储在优先队列中
        //该优先队列是一个小根堆，堆顶元素是所有牛栏右端点中最小的那个
        PriorityQueue<Pair> pq = new PriorityQueue<>();
        //遍历所有牛栏
        for(int i = 0; i < n; i ++)
        {
            if(pq.isEmpty() || pq.peek().tail >= interval[i].a)
            {
                int yardId = pq.size() + 1;
                ans[interval[i].id] = yardId;
                pq.add(new Pair(yardId, interval[i].b));
            }
            else
            {
                Pair top = pq.remove();
                ans[interval[i].id] = top.yardId;
                pq.add(new Pair(top.yardId, interval[i].b));
            }
        }
        bw.write(pq.size() + "\n");
        for(int i = 0; i < n; i ++)
        {
            bw.write(ans[i] + "\n");
        }
        bw.flush();
    }

    public static class Pair implements Comparable<Pair>
    {
        int yardId;
        int tail;
        public Pair(int yardId, int tail)
        {
            this.yardId = yardId;
            this.tail = tail;
        }
        @Override
        public int compareTo(Pair o)
        {
            //tail越小的排在越前面
            return this.tail - o.tail;
        }
    }

    //区间对象，存储区间的左右端点a和b
    //并且实现了Comparable接口，可以直接使用Arrays.sort()进行排序
    //排序规则为：先按照左端点a从小到大排序，如果a相同，则按照右端点b从小到大排序
    public static class Interval implements Comparable<Interval>
    {
        int id;
        int a, b;
        public Interval(int id, int a, int b)
        {
            this.id = id;
            this.a = a;
            this.b = b;
        }
        @Override
        public int compareTo(Interval o)
        {
            if(this.a != o.a)
            {
                return this.a - o.a;
            }
            else
            {
                return this.b - o.b;
            }
        }
    }
}
