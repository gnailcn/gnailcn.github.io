AcWing 137.雪花雪花雪花

2024-02-27 哈希 进阶指南 AcWing 算法

原题

有 N 片雪花，每片雪花由六个角组成，每个角都有长度。

第 i 片雪花六个角的长度从某个角开始顺时针依次记为 ai,1,ai,2,…,ai,6。

因为雪花的形状是封闭的环形，所以从任何一个角开始顺时针或逆时针往后记录长度，得到的六元组都代表形状相同的雪花。

例如 ai,1,ai,2,…,ai,6 和 ai,2,ai,3,…,ai,6，ai,1 就是形状相同的雪花。

ai,1,ai,2,…,ai,6 和 ai,6,ai,5,…,ai,1 也是形状相同的雪花。

我们称两片雪花形状相同，当且仅当它们各自从某一角开始顺时针或逆时针记录长度，能得到两个相同的六元组。

求这 N 片雪花中是否存在两片形状相同的雪花。

输入格式

第一行输入一个整数 N，代表雪花的数量。

接下来 N 行，每行描述一片雪花。

每行包含 6 个整数，分别代表雪花的六个角的长度（这六个数即为从雪花的随机一个角顺时针或逆时针记录长度得到）。

同行数值之间，用空格隔开。

输出格式

如果不存在两片形状相同的雪花，则输出：

No two snowflakes are alike.

如果存在两片形状相同的雪花，则输出：

Twin snowflakes found.

数据范围

1≤N≤100000,
0≤ai,j<10000000

输入样例：

2
1 2 3 4 5 6
4 3 2 1 6 5

输出样例：

Twin snowflakes found.

程序

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.Map;
import java.util.HashMap;
import java.util.List;
import java.util.ArrayList;

public class Main
{
    public static final int MOD = (int)1e9+7; //大质数
    public static final int N = (int)1e5+2;
    public static int[][] snows = new int[N][6];
    public static Map<Integer, List<Integer>> map = new HashMap<>();
    public static void main(String[] args) throws IOException
    {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        int n = Integer.parseInt(br.readLine());

        for(int i = 0; i < n; i ++)
        {
            String[] line = br.readLine().split(" ");
            for(int j = 0; j < 6; j ++)
            {
                int x = Integer.parseInt(line[j]);
                snows[i][j] = x;
            }
            int snowHash = getHash(snows[i]);
            add2map(snowHash, i);
        }
        if(check()) //存在相同的雪花
        {
            bw.write("Twin snowflakes found.");
        }
        else //不存在相同的雪花
        {
            bw.write("No two snowflakes are alike.");
        }
        bw.flush();
    }

    public static boolean check()
    {
        //对于每一个snowHash组内进行单独判断, 时间复杂度p*(n/p)*(n/p)*216 = n^2/p*216
        for(int snowHash: map.keySet())
        {
            List<Integer> list = map.get(snowHash);
            int len = list.size();
            for(int i = 0; i < len; i ++)
            {
                for(int j = i+1; j < len; j ++)
                {
                    int idxi = list.get(i);
                    int[] snow1 = snows[idxi];
                    int idxj = list.get(j);
                    int[] snow2 = snows[idxj];
                    if(cmp(snow1, snow2)) //存在相同雪花
                    {
                        return true;
                    }
                }
            }
        }
        return false; //不存在相同的雪花
    }

    public static boolean cmp(int[] snow1, int[] snow2)
    {
        for(int i = 0; i < 6; i ++) //snow1起点
        {
            for(int j = 0; j < 6; j ++) //snow2起点
            {
                boolean flag = true;
                //正序判断
                for(int k = 0; k < 6; k ++) //偏移
                {
                    if(snow1[(i+k)%6] != snow2[(j+k)%6])
                    {
                        flag = false;
                        break;
                    }
                }
                if(flag)
                {
                    return true;
                }
                flag = true;
                //逆序判断
                for(int k = 0; k < 6; k ++)
                {
                    if(snow1[(i+k)%6] != snow2[(j-k+6)%6])
                    {
                        flag = false;
                        break;
                    }
                }
                if(flag)
                {
                    return true;
                }
            }
        }
        return false;
    }

    public static void add2map(int snowHash, int idx)
    {
        if(!map.containsKey(snowHash))
        {
            List<Integer> list = new ArrayList<>();
            map.put(snowHash, list);
        }
        map.get(snowHash).add(idx);
    }

    public static int getHash(int[] snow)
    {
        //hash方式：在模MOD空间下，求和与求积
        int sum = 0;
        int mul = 1;
        for(int i = 0; i < 6; i ++)
        {
            sum = (sum + snow[i])%MOD;
            mul = (int)(((long)(mul)*snow[i])%MOD);
        }
        return (sum + mul)%MOD;
    }
}v
