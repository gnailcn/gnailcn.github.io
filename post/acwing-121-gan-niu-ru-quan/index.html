AcWing 121. 赶牛入圈

2023-12-19 离散化 进阶指南 AcWing 二分 前缀和 算法

原题

农夫约翰希望为他的奶牛们建立一个畜栏。

这些挑剔的畜生要求畜栏必须是正方形的，而且至少要包含 C 单位的三叶草，来当做它们的下午茶。

畜栏的边缘必须与 X，Y 轴平行。

约翰的土地里一共包含 N 单位的三叶草，每单位三叶草位于一个 1×1 的土地区域内，区域位置由其左下角坐标表示，并且区域左下角的 X,Y 坐标都为整数，范围在 1 到 10000 以内。

多个单位的三叶草可能会位于同一个 1×1 的区域内，因为这个原因，在接下来的输入中，同一个区域坐标可能出现多次。

只有一个区域完全位于修好的畜栏之中，才认为这个区域内的三叶草在畜栏之中。

请你帮约翰计算一下，能包含至少 C 单位面积三叶草的情况下，畜栏的最小边长是多少。

输入格式

第一行输入两个整数 C 和 N。

接下来 N 行，每行输入两个整数 X 和 Y，代表三叶草所在的区域的 X,Y 坐标。

同一行数据用空格隔开。

输出格式

输出一个整数，代表畜栏的最小边长。

数据范围

1≤C≤500,
C≤N≤500

输入样例：

3 4
1 2
2 1
4 1
5 2

输出样例：

4

import java.util.Scanner;
import java.util.Set;
import java.util.TreeSet; // 有序集合

public class Main
{
    public static final int N = 505;
    public static Node[] nodes = new Node[N];
    public static int c;
    public static int n;
    public static Set<Integer> set = new TreeSet<Integer>();
    public static int setLen;
    //注意id2val从1开始
    public static int[] id2val = new int[N*2]; //需要离散化的元素有n*2个
    public static int[][] sum;
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        c = sc.nextInt();
        n = sc.nextInt();
        for(int i = 0; i < n; i ++)
        {
            int x = sc.nextInt();
            int y = sc.nextInt();
            nodes[i] = new Node(x, y);
            set.add(x);
            set.add(y);
        }
        //离散化
        setLen = 0;
        for(int value: set)
        {
            setLen ++; //从1开始，因为后面需要计算前缀和
            id2val[setLen] = value;
        }
        sum = new int[setLen + 1][setLen + 1];
        //将Node加到sum中
        for(int i = 0; i < n; i ++)
        {
            int x = nodes[i].x;
            int y = nodes[i].y;
            sum[getIdxbyVal(x)][getIdxbyVal(y)] ++;
        }
        //计算前缀和
        for(int i = 1; i <= setLen; i ++)
        {
            for(int j = 1; j <= setLen; j ++)
            {
                sum[i][j] += sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1];
            }
        }
        //二分正方形的边长
        int l = 1, r = 10000;
        while(l < r)
        {
            int mid = (l + r) >> 1;
            if(check(mid)) r = mid;
            else l = mid + 1;
        }
        System.out.println(l);
    }

    //根据数的val查找id
    //主要找的是小于等于val的最大值的id
    //注意不存在所有的元素都比val大的情况（这是由后续任务的写法决定的）
    //所以在1~setLen中一定存在一个id，使得id2val[id] <= val并且id2val[id+1] > val
    public static int getIdxbyVal(int val)
    {
        //二分查找
        int l = 1, r = setLen;
        while(l < r)
        {
            int mid = (l + r + 1) >> 1;
            if(id2val[mid] <= val) l = mid;
            else r = mid - 1;
        }
        return l;
    }

    public static boolean check(int len)
    {
        //查看是否存在正方形的边长为len
        //满足里面的元素个数>=c
        for(int i = 1; i <= setLen; i ++)
        {
            for(int j = 1; j <= setLen; j ++)
            {
                int x1 = i;
                int y1 = j;
                int valX = id2val[x1];
                int valY = id2val[y1];
                int x2 = getIdxbyVal(valX + len - 1);
                int y2 = getIdxbyVal(valY + len - 1);
                int cnt = sum[x2][y2] - sum[x1 - 1][y2] - sum[x2][y1 - 1] + sum[x1 - 1][y1 - 1];
                if(cnt >= c) return true;
            }
        }
        return false;
    }

    public static class Node
    {
        public int x, y;
        public Node(int x, int y)
        {
            this.x = x;
            this.y = y;
        }
    }
}
