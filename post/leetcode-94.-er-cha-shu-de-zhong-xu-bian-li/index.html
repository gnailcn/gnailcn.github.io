leetcode 94. 二叉树的中序遍历

2023-10-22 leetcode 树 算法

给定一个二叉树的根节点 root ，返回 它的 中序 遍历 。

示例 1：

输入：root = [1,null,2,3]
输出：[1,3,2]

示例 2：

输入：root = []
输出：[]

示例 3：

输入：root = [1]
输出：[1]

提示：

  • 树中节点数目在范围 [0, 100] 内
  • -100 <= Node.val <= 100

进阶： 递归算法很简单，你可以通过迭代算法完成吗？

程序：

  • 特殊迭代法
import java.util.Stack;
import java.util.List;
import java.util.ArrayList;

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        Stack<TreeNode> S = new Stack<TreeNode>();
        List<Integer> ans = new ArrayList<Integer>();
        TreeNode cur = root;
        while(cur!=null || S.size()>0)
        {
            //一直往左
            while(cur!=null)
            {
                S.push(cur);
                cur = cur.left;
            }
            //cur为空节点
            TreeNode node = S.pop();
            ans.add(node.val);
            cur = node.right;
        }
        return ans;
    }
}
  • 一般迭代法
import java.util.List;
import java.util.ArrayList;
import java.util.Stack;

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        Stack<TreeNode> S = new Stack<TreeNode>();
        List<Integer> ans = new ArrayList<Integer>();
        if(root==null)
        {
            return ans;
        }
        S.add(root);
        while(S.size()>0)
        {
            TreeNode top = S.pop();
            if(top==null)
            {
                //遇到标记，说明当前的栈顶元素被访问过，直接回溯处理数据
                TreeNode node = S.pop();
                ans.add(node.val);
            }
            else
            {
                //访问top节点的子节点, 并且标记top节点等待回溯
                //右中左
                if(top.right!=null) S.push(top.right);
                S.push(top);
                S.push(null); //标记，说明访问过了
                if(top.left!=null)  S.push(top.left);
            }
        }
        return ans;
    }
}
